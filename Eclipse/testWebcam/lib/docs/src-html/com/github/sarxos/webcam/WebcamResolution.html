001    package com.github.sarxos.webcam;
002    
003    import java.awt.Dimension;
004    
005    
006    public enum WebcamResolution {
007    
008            /**
009             * 176x144
010             */
011            QQVGA(176, 144),
012    
013            /**
014             * 320x240
015             */
016            QVGA(320, 240),
017    
018            /**
019             * 352x288
020             */
021            CIF(352, 288),
022    
023            HVGA(480, 400),
024            VGA(640, 480),
025            PAL(768, 576),
026            SVGA(800, 600),
027    
028            /**
029             * 1024x768
030             */
031            XGA(1024, 768),
032    
033            /**
034             * 1280x720
035             */
036            HD720(1280, 720),
037            WXGA(1280, 768),
038            SXGA(1280, 1024),
039            UXGA(1600, 1200),
040            QXGA(2048, 1536);
041    
042            private Dimension size = null;
043    
044            private WebcamResolution(int width, int height) {
045                    this.size = new Dimension(width, height);
046            }
047    
048            public Dimension getSize() {
049                    return size;
050            }
051    }
